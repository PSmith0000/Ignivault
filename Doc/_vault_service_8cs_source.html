134 public async Task<bool> UpdateItemAsync(int itemId, UpdateVaultItemDto itemDto, string userId)
135 {
136 var existingEntity = await _vaultItemRepository.GetItemByIdAndUserIdAsync(itemId, userId);
137 if (existingEntity == null)
138 {
139 return false;
140 }
141
142 existingEntity.Name = itemDto.Name;
143 if (itemDto.EncryptedData != null && itemDto.Iv != null)
144 {
145 existingEntity.EncryptedData = itemDto.EncryptedData;
146 existingEntity.Iv = itemDto.Iv;
147 }
148
149 // Manually update the timestamp to reflect the change.
150 existingEntity.UpdatedAt = DateTime.UtcNow;
151
152 var success = await _vaultItemRepository.UpdateItemAsync(existingEntity);
153
154 if (success)
155 {
156 await _userActivityLogger.LogActivityAsync(userId, ActivityType.VaultItemUpdated, true, $"Updated item: {existingEntity.Name} (ID: {existingEntity.ItemId})");
157 }
158
159 return success;
160 }
